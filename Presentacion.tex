\section{Objetivos}
\newtheorem{teo2-a}{Objetivo General}
\newtheorem{teo2-b}{Objetivos Específicos}
\subsection{Objetivo General}
\begin{frame}
	\frametitle{Objetivos General}
	\begin{teo2-a}
		Diseñar e implementar un prototipo de software-hardware basado en un microcontrolador Arduino y un sensor de velocidad angular y acelerometría de 3 ejes, para el registro y representación gráfica del centro de masa y bio-realimentación..
	\end{teo2-a}	
\end{frame}

\subsection{Objetivos Específicos}
\begin{frame}
	\frametitle{Objetivos Específicos}
	\begin{teo2-b}
		\begin{itemize}
		\small \item Integrar microcontrolador Arduino con sensor (giroscopio-acelerómetro).\pause
	   	\item Diseñar sistema que permita el registro y visualización de todas las variables cinemáticas
	   	(posición y velocidad angular) del centro de masa.\pause
	   	\item Construcción de un sistema que facilite mediante bio-realimentación la posición del centro de presión (proyección del centro de masa).
	   	\end{itemize}
	\end{teo2-b}
\end{frame}
